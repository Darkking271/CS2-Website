  1 /*
  2  * This program will read a text file for a song or poem, and print words longer
  3  * than average
  4  */
  5 package TextProcessing;
  6 
  7 import java.io.FileReader;
  8 import java.io.BufferedReader;
  9 import java.util.Scanner;
 10 import java.util.ArrayList;
 11 
 12 /**
 13  *
 14  * @author Alex
 15  */
 16 public class LongerWords 
 17 {
 18 
 19     /**
 20      * @param args the command line arguments
 21      * @throws java.lang.Exception
 22      */
 23     public static void main(String[] args) throws Exception
 24     {
 25         Scanner keyboard = new Scanner(System.in);
 26         String[] words;
 27         String in;
 28         
 29         System.out.print("Lyrics or Poem?:  ");
 30         in = keyboard.nextLine();
 31         in = in.toLowerCase();
 32         System.out.println();
 33         
 34         switch (in)
 35         {
 36             case "lyrics":
 37                 recieveFile("Flickering Flame.txt");
 38                 break;
 39             case "poem": 
 40                 recieveFile("The Wicked.txt");
 41                 break;
 42         }//END SWITCH
 43         
 44         
 45     }//END MAINE
 46 
 47     private static void recieveFile(String link) throws Exception
 48     {
 49         FileReader file = new FileReader("C:/Users/Alex/Documents/School/2016 Fall/CSC-241/CS2F16/Text Documents/" + link);
 50         BufferedReader reader = new BufferedReader(file);
 51         String line = reader.readLine();
 52         String text = "";
 53         
 54         while (line != null)
 55         {
 56             text += line;
 57             line = reader.readLine();
 58         }//END WHILE
 59         
 60         String[] words = text.split(" ");
 61         ArrayList<String> longest = longestWords(words);
 62         display(longest);
 63     }
 64 
 65     private static ArrayList<String> longestWords(String[] list) 
 66     {  
 67         int mean;
 68         int total = 0;
 69         ArrayList<String> longer = new ArrayList<>();
 70         
 71         for (String list1 : list) 
 72         {
 73             total += list1.length();
 74         } //END FOR
 75         
 76         mean = total / list.length;
 77         
 78         for (String list1 : list) 
 79         {
 80             if (list1.length() > mean) 
 81                 longer.add(list1);
 82         } //END FOR
 83         
 84         return longer;
 85         
 86     }//END longestWords
 87 
 88     private static void display(ArrayList<String> list) 
 89     {
 90         System.out.println("The longer words...\n");
 91         
 92         for (int i = 0; i < list.size(); i++)
 93         {
 94             System.out.print(list.get(i) + " ");
 95             
 96             if ((i + 1) % 10 == 0)
 97                 System.out.println();
 98         }//END FOR
 99         
100         System.out.println();
101     }
102     
103 }//END LONGER WORDS

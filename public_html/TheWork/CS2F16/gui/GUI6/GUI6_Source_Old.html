 4 package frames;

 6 import java.awt.BorderLayout;
 7 import java.awt.Color;
 8 import java.awt.Container;
 9 import java.awt.event.ActionEvent;
10 import java.awt.event.ActionListener;
11 import javax.swing.JButton;
12 import javax.swing.JFrame;
13 import javax.swing.JPanel;
14 
15 /**
16  *
17  * @author Alex
18  */
19     //modeling the feature frame of the GUI
20 public class KFrame6 extends JFrame implements ActionListener 
21 {
22  private JButton blueButton;
23  private JButton redButton;
24  private JButton greenButton;
25  private JButton yellowButton;
26  private JPanel reflector;
27  
28  public KFrame6(String title)
29  {
30      super(title);
31      setSize(500, 300);
32      setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
33      addComponents(getContentPane());
34      addListeners();
35      setVisible(true);
36      
37  }//END KFrame6
38  
39  private void addComponents(Container contentPane)
40  {
41      blueButton = new JButton("Blue");
42      redButton = new JButton("Red");
43      greenButton = new JButton("Green");
44      yellowButton = new JButton("Yellow");
45      reflector = new JPanel();
46      contentPane.setLayout(new BorderLayout());
47      contentPane.add(blueButton, BorderLayout.NORTH);
48      contentPane.add(redButton, BorderLayout.SOUTH);
49      contentPane.add(greenButton, BorderLayout.EAST);
50      contentPane.add(yellowButton, BorderLayout.WEST);
51      contentPane.add(reflector, BorderLayout.CENTER);
52      
53  }//END addComponents
54  
55  //Add the listeners to the frame.
56  private void addListeners()
57  {
58      blueButton.addActionListener(this);
59      redButton.addActionListener(this);
60      greenButton.addActionListener(this);
61      yellowButton.addActionListener(this);
62      
63  }//END addListeners
64  
65  //This method serves to implement the ActionListener interface
66  public void actionPerformed(ActionEvent event)
67  {
68      String command = event.getActionCommand();
69      if (command.equals("Red"))
70          reflector.setBackground(Color.RED);
71      else if (command.equals("Yellow"))
72          reflector.setBackground(Color.YELLOW);
73      else if (command.equals("Blue"))
74          reflector.setBackground(Color.BLUE);
75      else if (command.equals("Green"))
76          reflector.setBackground(Color.GREEN);
77  }
78 }//END KFrame6

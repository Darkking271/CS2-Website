  6 package frames;
  7 
  8 import java.awt.BorderLayout;
  9 import java.awt.Container;
 10 import java.awt.FlowLayout;
 11 import java.awt.event.ActionEvent;
 12 import java.awt.event.ActionListener;
 13 import javax.swing.JButton;
 14 import javax.swing.JFrame;
 15 import javax.swing.JPanel;
 16 import javax.swing.JTextArea;
 17 import javax.swing.JTextField;
 18 
 19 /**
 20  *
 21  * @author Alex
 22  */
 23 public class KFrame8p extends JFrame implements ActionListener 
 24 {    
 25     JTextArea output;
 26     JButton pointButton;
 27     JButton lineButton;
 28     JButton wordButton;
 29     JButton quoteButton;
 30     JButton colorButton;
 31     JButton clearButton;
 32     JTextField input;
 33     
 34     public KFrame8p(String title)
 35     {
 36         super(title);
 37         setSize(500, 300);
 38         setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 39         addComponents(getContentPane());
 40         addListeners();
 41         setVisible(true);
 42         
 43     }//END KFrame7
 44         
 45     private void addComponents(Container contentPane)
 46     {
 47         //components for northern region
 48         pointButton = new JButton("Point");
 49         lineButton = new JButton("Line");
 50         wordButton = new JButton("Word");
 51         quoteButton = new JButton("Quote");
 52         colorButton = new JButton("Color");
 53         clearButton = new JButton("Clear");
 54         JPanel controlPanel = new JPanel();
 55         controlPanel.setLayout(new FlowLayout());
 56         controlPanel.add(pointButton);
 57         controlPanel.add(lineButton);
 58         controlPanel.add(wordButton);
 59         controlPanel.add(quoteButton);
 60         controlPanel.add(colorButton);
 61         controlPanel.add(clearButton);
 62         
 63         //components central region
 64         output = new JTextArea();
 65         
 66         //components for southern region
 67         input = new JTextField();
 68         
 69         //establish the regions
 70         contentPane.setLayout(new BorderLayout());
 71         contentPane.add(controlPanel, BorderLayout.NORTH);
 72         contentPane.add(output, BorderLayout.CENTER);
 73         contentPane.add(input, BorderLayout.SOUTH);
 74         
 75     }//END addComponents
 76     
 77     //Add the listeners to the frame.
 78     private void addListeners()
 79     {
 80         pointButton.addActionListener(this);
 81         lineButton.addActionListener(this);
 82         wordButton.addActionListener(this);
 83         quoteButton.addActionListener(this);
 84         colorButton.addActionListener(this);
 85         clearButton.addActionListener(this);
 86         input.addActionListener(this);
 87         
 88     }//END addListeners
 89     
 90     //This method serves to implement the ActionListener interface
 91     public void actionPerformed(ActionEvent event)
 92     {
 93         if (event.getSource() instanceof JTextField)
 94             input.setText("");
 95         
 96         switch(event.getActionCommand().toUpperCase())
 97         {
 98             case "POINT":
 99                 newLine("POINT");
100                 break;
101             case "LINE":
102                 newLine("LINE");
103                 break;
104             case "WORD":
105                 newLine("WORD");
106                 break;
107             case "QUOTE":
108                 newLine("QUOTE");
109                 break;
110             case "COLOR":
111                 newLine("COLOR");
112                 break;
113             case "CLEAR":
114                 newLine("CLEAR");
115                 break;
116             case "CL":              //secret clear functionality
117                 output.setText("");
118                 break;
119                 
120         }//END SWITCH
121         
122     }//END actionPerformed
123 
124     private void newLine(String word) 
125     {
126         output.append(word + "\n");
127     }
128        
129 
130 }//END KFrame8p

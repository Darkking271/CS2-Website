  7 package gui;

  9 import java.awt.BorderLayout;
 10 import java.awt.Color;
 11 import java.awt.Container;
 12 import java.awt.event.ActionEvent;
 13 import java.awt.event.ActionListener;
 14 import javax.swing.JButton;
 15 import javax.swing.JFrame;
 16 import javax.swing.JPanel;
 17 import javax.swing.SwingUtilities;
 18 
 19 /**
 20  *
 21  * @author Alex
 22  */
 23 public class GUI5 {
 24 
 25     /**
 26      * @param args the command line arguments
 27      */
 28     public static void main(String[] args) 
 29     {
 30         SwingUtilities.invokeLater(new ThreadForGUI());
 31         
 32     }//END MAIN
 33     
 34     private static class ThreadForGUI implements Runnable
 35     {
 36         @Override
 37         public void run()
 38         {
 39             GUI5 gui = new GUI5();
 40             
 41         }//END RUN
 42         
 43     }//END ThreadForGUI
 44     
 45     public GUI5()
 46     {
 47         KFrame frame = new KFrame("GUI 5");
 48         
 49     }//END GUI5
 50     
 51     //modeling the feature fram of the GUI
 52     
 53     public class KFrame extends JFrame implements ActionListener
 54     {
 55          private JButton blueButton;
 56          private JButton redButton;
 57          private JButton greenButton;
 58          private JButton yellowButton;
 59          private JPanel reflector;
 60          
 61          public KFrame (String title)
 62          {
 63              super(title);
 64              setSize(500,300);
 65              setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 66              addComponents(getContentPane());
 67              addListeners();
 68              setVisible(true);
 69              
 70          }//END KFame
 71          
 72          private void addComponents(Container contentPane)
 73          {
 74              blueButton = new JButton("Blue");
 75              redButton = new JButton("Red");
 76              greenButton = new JButton("Green");
 77              yellowButton = new JButton("Yellow");
 78              reflector = new JPanel();
 79              contentPane.setLayout(new BorderLayout());
 80              contentPane.add(blueButton, BorderLayout.NORTH);
 81              contentPane.add(redButton, BorderLayout.SOUTH);
 82              contentPane.add(greenButton, BorderLayout.EAST);
 83              contentPane.add(yellowButton, BorderLayout.WEST);
 84              contentPane.add(reflector, BorderLayout.CENTER);
 85              
 86          }//END addComponents
 87          
 88          //Add the listeners to the frame
 89          
 90          private void addListeners()
 91          {
 92              blueButton.addActionListener(this);
 93              redButton.addActionListener(this);
 94              greenButton.addActionListener(this);
 95              yellowButton.addActionListener(this);
 96              
 97          }//END addListeners
 98          
 99          //This method serves to implement the ActionListener interface
100          
101          @Override
102          public void actionPerformed(ActionEvent event)
103          {
104              String command = event.getActionCommand();
105              if (command.equals("Red"))
106                  reflector.setBackground(Color.RED);
107              else if (command.equals("Yellow"))
108                  reflector.setBackground(Color.YELLOW);
109              else if (command.equals("Blue"))
110                  reflector.setBackground(Color.BLUE);
111              else if (command.equals("Green"))
112                  reflector.setBackground(Color.GREEN);
113          }//END actionPerformed
114          
115     }//END Frame
116     
117 }//END GUI5

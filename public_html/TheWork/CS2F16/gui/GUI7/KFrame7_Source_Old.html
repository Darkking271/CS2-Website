 24 public class KFrame7 extends JFrame implements ActionListener
 25 {
 26     private JPanel reflector;
 27     private JButton blueButton;
 28     private JButton redButton;
 29     private JButton greenButton;
 30     private JButton yellowButton;
 31     private JButton colorButton;
 32     private JTextField input;
 33     
 34     public KFrame7(String title)
 35     {
 36         super(title);
 37         setSize(500, 300);
 38         setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 39         addComponents(getContentPane());
 40         addListeners();

 42         
 43     }//END KFrame7
 44     
 45     private void addComponents(Container contentPane)
 46     {
 47         //components for northern region
 48         blueButton = new JButton("Blue");
 49         redButton = new JButton("Red");
 50         greenButton = new JButton("Green");
 51         yellowButton = new JButton("Yellow");
 52         colorButton = new JButton("Color");
 53         JPanel controlPanel = new JPanel();
 54         controlPanel.setLayout(new FlowLayout());
 55         controlPanel.add(blueButton);
 56         controlPanel.add(redButton);
 57         controlPanel.add(greenButton);
 58         controlPanel.add(yellowButton);
 59         controlPanel.add(colorButton);
 60         
 61         //components central region
 62         reflector = new JPanel();
 63         
 64         //components for southern region
 65         input = new JTextField();
 66         
 67         //establish the regions
 68         contentPane.setLayout(new BorderLayout());
 69         contentPane.add(controlPanel, BorderLayout.NORTH);
 70         contentPane.add(reflector, BorderLayout.CENTER);
 71         contentPane.add(input, BorderLayout.SOUTH);
 72         
 73     }//END addComponents

 75     //Add the listeners to the frame.
 76     private void addListeners()
 77     {
 78         blueButton.addActionListener(this);
 79         redButton.addActionListener(this);
 80         greenButton.addActionListener(this);
 81         yellowButton.addActionListener(this);
 82         colorButton.addActionListener(this);
 83         input.addActionListener(this);
 84         
 85     }//END addListeners

 87     //This method serves to implement the ActionListener interface
 88     public void actionPerformed(ActionEvent event)
 89     {
 90         String command = event.getActionCommand();
 91         if (event.getSource() instanceof JTextField)
 92             input.setText("");
 93         
 94         switch(event.getActionCommand().toUpperCase())
 95         {
 96             case "RED":
 97                 reflector.setBackground(Color.RED);
 98                 break;
 99             case "YELLOW":
100                 reflector.setBackground(Color.YELLOW);
101                 break;
102             case "BLUE":
103                 reflector.setBackground(Color.BLUE);
104                 break;
105             case "GREEN":
106                 reflector.setBackground(Color.GREEN);
107                 break;
108             case "COLOR":
109                 reflector.setBackground(Random.color());
110                 break;
111                 
112         }//END SWITCH
113         
114     }//END actionPerformed

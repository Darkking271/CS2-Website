  7 import java.awt.BorderLayout;
  8 import java.awt.Color;
  9 import java.awt.Container;
 10 import java.awt.FlowLayout;
 11 import java.awt.event.ActionEvent;
 12 import java.awt.event.ActionListener;
 13 import javax.swing.JButton;
 14 import javax.swing.JFrame;
 15 import javax.swing.JPanel;
 16 import javax.swing.JTextField;
 17 import utilities.Random;
 18 /**
 19  *
 20  * @author Alex
 21  */
 22     //modeling the feature frame for the GUI
 23 
 24 public class KFrame7 extends JFrame implements ActionListener
 25 {
 26     private JPanel reflector;
 27     private JButton blueButton;
 28     private JButton redButton;
 29     private JButton greenButton;
 30     private JButton yellowButton;
 31     private JButton colorButton;
 32     private JTextField input;
 33     Random gen = new Random();
 34     
 35     public KFrame7(String title)
 36     {
 37         super(title);
 38         setSize(500, 300);
 39         setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 40         addComponents(getContentPane());
 41         addListeners();
 42         setVisible(true);
 43         
 44     }//END KFrame7
 45     
 46     private void addComponents(Container contentPane)
 47     {
 48         //components for northern region
 49         blueButton = new JButton("Blue");
 50         redButton = new JButton("Red");
 51         greenButton = new JButton("Green");
 52         yellowButton = new JButton("Yellow");
 53         colorButton = new JButton("Color");
 54         JPanel controlPanel = new JPanel();
 55         controlPanel.setLayout(new FlowLayout());
 56         controlPanel.add(blueButton);
 57         controlPanel.add(redButton);
 58         controlPanel.add(greenButton);
 59         controlPanel.add(yellowButton);
 60         controlPanel.add(colorButton);
 61         
 62         //components central region
 63         reflector = new JPanel();
 64         
 65         //components for southern region
 66         input = new JTextField();
 67         
 68         //establish the regions
 69         contentPane.setLayout(new BorderLayout());
 70         contentPane.add(controlPanel, BorderLayout.NORTH);
 71         contentPane.add(reflector, BorderLayout.CENTER);
 72         contentPane.add(input, BorderLayout.SOUTH);
 73         
 74     }//END addComponents
 75     
 76     //Add the listeners to the frame.
 77     private void addListeners()
 78     {
 79         blueButton.addActionListener(this);
 80         redButton.addActionListener(this);
 81         greenButton.addActionListener(this);
 82         yellowButton.addActionListener(this);
 83         colorButton.addActionListener(this);
 84         input.addActionListener(this);
 85         
 86     }//END addListeners
 87     
 88     //This method serves to implement the ActionListener interface
 89     public void actionPerformed(ActionEvent event)
 90     {
 91         String command = event.getActionCommand();
 92         if (event.getSource() instanceof JTextField)
 93             input.setText("");
 94         
 95         switch(event.getActionCommand().toUpperCase())
 96         {
 97             case "RED":
 98                 reflector.setBackground(Color.RED);
 99                 break;
100             case "YELLOW":
101                 reflector.setBackground(Color.YELLOW);
102                 break;
103             case "BLUE":
104                 reflector.setBackground(Color.BLUE);
105                 break;
106             case "GREEN":
107                 reflector.setBackground(Color.GREEN);
108                 break;
109             case "COLOR":
110                 reflector.setBackground(Random.color());
111                 break;
112                 
113         }//END SWITCH
114         
115     }//END actionPerformed
116     
117 }//END KFrame7

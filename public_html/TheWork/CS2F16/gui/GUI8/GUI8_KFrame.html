  6 import java.awt.BorderLayout;
  7 import java.awt.Container;
  8 import java.awt.FlowLayout;
  9 import java.awt.event.ActionEvent;
 10 import java.awt.event.ActionListener;
 11 import java.awt.Color;
 12 import javax.swing.JButton;
 13 import javax.swing.JFrame;
 14 import javax.swing.JPanel;
 15 import javax.swing.JTextArea;
 16 import javax.swing.JTextField;
 17 import utilities.Random;

 19 /**
 20  *
 21  * @author Alex
 22  */
 23 public class KFrame8 extends JFrame implements ActionListener 
 24 {    
 25     JTextArea output;
 26     JButton pointButton;
 27     JButton lineButton;
 28     JButton wordButton;
 29     JButton quoteButton;
 30     JButton colorButton;
 31     JButton clearButton;
 32     JTextField input;
 33     Random gen = new Random();
 34     
 35     public KFrame8(String title)
 36     {
 37         super(title);
 38         setSize(500, 600);
 39         setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 40         addComponents(getContentPane());
 41         addListeners();
 42         setVisible(true);
 43         
 44     }//END KFrame7
 45         
 46     private void addComponents(Container contentPane)
 47     {
 48         //components for northern region
 49         pointButton = new JButton("Point");
 50         lineButton = new JButton("Line");
 51         wordButton = new JButton("Word");
 52         quoteButton = new JButton("Quote");
 53         colorButton = new JButton("Color");
 54         clearButton = new JButton("Clear");
 55         JPanel controlPanel = new JPanel();
 56         controlPanel.setLayout(new FlowLayout());
 57         controlPanel.add(pointButton);
 58         controlPanel.add(lineButton);
 59         controlPanel.add(wordButton);
 60         controlPanel.add(quoteButton);
 61         controlPanel.add(colorButton);
 62         controlPanel.add(clearButton);
 63         
 64         //components central region
 65         output = new JTextArea();
 66         
 67         //components for southern region
 68         input = new JTextField();
 69         
 70         //establish the regions
 71         contentPane.setLayout(new BorderLayout());
 72         contentPane.add(controlPanel, BorderLayout.NORTH);
 73         contentPane.add(output, BorderLayout.CENTER);
 74         contentPane.add(input, BorderLayout.SOUTH);
 75         
 76     }//END addComponents
 77     
 78     //Add the listeners to the frame.
 79     private void addListeners()
 80     {
 81         pointButton.addActionListener(this);
 82         lineButton.addActionListener(this);
 83         wordButton.addActionListener(this);
 84         quoteButton.addActionListener(this);
 85         colorButton.addActionListener(this);
 86         clearButton.addActionListener(this);
 87         input.addActionListener(this);
 88         
 89     }//END addListeners
 90     
 91     //This method serves to implement the ActionListener interface
 92     public void actionPerformed(ActionEvent event)
 93     {
 94         if (event.getSource() instanceof JTextField)
 95             input.setText("");
 96         
 97         switch(event.getActionCommand().toUpperCase())
 98         {
 99             case "POINT":
100                 newPoint();
101                 break;
102             case "LINE":
103                 newPointLine();
104                 break;
105             case "WORD":
106                 newWord();
107                 break;
108             case "QUOTE":
109                 newQuote();
110                 break;
111             case "COLOR":
112                 output.setBackground(Random.color());
113                 break;
114             case "CLEAR":
115                 output.setText("");
116                 output.setBackground(Color.WHITE);
117                 break;
118                                 
119         }//END SWITCH
120         
121     }//END actionPerformed
122 
123     private void newLine(String word) 
124     {
125         output.append(word + "\n\n");
126         
127     }//END newLine
128     
129     private void newPoint()

131         newLine(Random.point(500, 600));
132         
133     }//END newPoint
134     
135     private void newPointLine()
136     {
137         newLine(Random.point(500, 600) + 
138                 " " + Random.line(600, 359));
139         
140     }//END newPointLine
141     
142     private void newWord()
143     {
144         newLine(gen.word());
145         
146     }//END newWord
147        
148     private void newQuote()
149     {
150         newLine(gen.quote());
151         
152     }//END newQuote

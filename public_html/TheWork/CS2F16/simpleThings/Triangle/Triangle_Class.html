10 public class Triangle implements TriangleADT
11 {
12     private double sideOne;
13     private double sideTwo;
14     private double sideThree;
15     
16     public Triangle(int side1, int side2, int side3)
17     {
18         sideOne = side1;
19         sideTwo = side2;
20         sideThree = side3;
21         
22     }//END Triangle
23     
24     public Triangle(int sides)
25     {
26         sideOne = sides;
27         sideTwo = sides;
28         sideThree = sides;
29         
30     }//END Triangle Overloaded
31     
32     @Override
33     public String toString()
34     {
35         String type = "<Triangle with A = " + sideOne + 
36                       ", B = " + sideTwo + 
37                       ", C = " + sideThree + ">";
38         return type;
39     }
40     
41     @Override
42     public double perimeter()
43     {
44         double perimeter = sideOne + sideTwo + sideThree;
45         return perimeter;
46         
47     }//END perimeter
48     
49     @Override
50     public double area()
51     {
52         double p = (sideOne + sideTwo + sideThree) / 2;
53         double area = Math.sqrt(p * (p - sideOne) * (p - sideTwo) * (p - sideThree));
54         return area;
55         
56     }//END area
57     
58 }//END Triangle Class

 28     private static void testLR() 
 29     {
 30         System.out.println("Test LR ...");
 31         BinaryTree<String, String> t = new BinaryTree<String, String>();
 32         System.out.println(">>> Creating empty tree.");
 33         System.out.println(">>> Adding elements to the tree directionally ...");
 34         try
 35         {
 36             t.addLR("Red", null, "");
 37             t.addLR("Blue", null, "l");
 38             t.addLR("Green", null, "lr");
 39             t.addLR("Black", null, "ll");
 40             t.addLR("White", null, "r");
 41             t.addLR("Purple", null, "lrr");
 42             t.addLR("Yellow", null, "rr");
 43             t.addLR("Orange", null, "lrrl");
 44             t.addLR("Indigo", null, "lrrll");
 45             t.addLR("Aqua", null, "lll");
 46             t.addLR("Tangerine", null, "rl");
 47             t.addLR("Pink", null, "lrrlr");
 48             t.addLR("Ghostwhite", null, "lrl");
 49             t.addLR("Gray", null, "lrlr");
 50             t.addLR("Brown", null, "lrll");
 51             t.addLR("Gainsboro", null, "lrllr");
 52         }
 53         catch (BinaryTreeCreationException ex)
 54         {
 55             ex.printStackTrace();
 56         }
 57         System.out.println(">>> PREORDER");
 58         t.preorder();
 59         System.out.println(">>> INORDER");
 60         t.inorder();
 61         System.out.println(">>> POSTORDER");
 62         t.postorder();
 63         System.out.println(">>> HEIGHT = " + t.height());
 64         System.out.println("\n");
 65     }//END testLR

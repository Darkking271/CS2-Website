46     private static ArrayList<String> getWords(Scanner scanner)

56     private static BinaryTree<String, Integer> createTree(ArrayList<String> words) 
57     throws BinaryTreeCreationException 
58     {
59         BinaryTree<String, Integer> tree = new BinaryTree<>();
60         for (int i = 0; i < words.size(); i++)
61         {
62             if (tree.member(words.get(i)))
63             {
64                 BinaryTree<String, Integer> node = tree.find(words.get(i));
65                 node.setValue(node.value() + 1);
66             }
67             else
68                 tree.addST(words.get(i),1);
69         }
70         return tree;
71     }

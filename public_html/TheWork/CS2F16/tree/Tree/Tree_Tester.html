 26     private static void testLR() 
 27     {
 28         System.out.println("Test LR ...");
 29         BinaryTree<String, String> t = new BinaryTree<String, String>();
 30         System.out.println(">>> Creating empty tree.");
 31         System.out.println(">>> Adding elements to the tree directionally ...");
 32         try
 33         {
 34             t.addLR("lion", null, "");
 35             t.addLR("shark", null, "r");
 36             t.addLR("elephant", null, "l");
 37             t.addLR("zebra", null, "rr");
 38             t.addLR("alligator", null, "ll");
 39             t.addLR("ardvaark", null, "llr");
 40             t.addLR("tiger", null, "rrl");
 41             t.addLR("giraffe", null, "lr");
 42             t.addLR("bear", null, "llrr");
 43             t.addLR("snake", null, "rrll");
 44         }
 45         catch (BinaryTreeCreationException ex)
 46         {
 47             ex.printStackTrace();
 48         }
 49         System.out.println(">>> PREORDER");
 50         t.preorder();
 51         System.out.println(">>> INORDER");
 52         t.inorder();
 53         System.out.println(">>> POSTORDER");
 54         t.postorder();
 55         System.out.println(">>> HEIGHT = " + t.height());
 56         System.out.println("\n");
 57     }//END testLR

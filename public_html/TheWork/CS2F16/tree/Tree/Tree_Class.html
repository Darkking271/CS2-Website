  4 package trees;

  6 /**
  7  *
  8  * @author Alex
  9  */
 10 public class BinaryTree<K, V> implements BinaryTreeADT<K, V>
 11 {
 12     private K key;
 13     private V value;
 14     private BinaryTree<K, V> lchild;
 15     private BinaryTree<K, V> rchild;
 16 
 17     public BinaryTree()
 18     {
 19         key = null;
 20     }//END BinaryTree constructor
 21     
 22     public V value() 
 23     {
 24         return value;
 25     }//END value
 26 
 27     public void setValue(V value) 
 28     {
 29         this.value = value;
 30     }//END setValue
 31 
 32     public boolean empty() 
 33     {
 34         return key == null;
 35     }//END empty
 36 
 37     public int height() 
 38     {
 39         if (empty())
 40             return 0;
 41         else
 42             return 1 + max(lchild.height(), rchild.height());
 43     }//END height
 44     
 45     private int max(int a, int b)
 46     {
 47         if (a > b)
 48             return a;
 49         else
 50             return b;
 51     }//END max
 52 
 53     public void preorder() 
 54     {
 55         if (!empty())
 56         {
 57             visit();
 58             lchild.preorder();
 59             rchild.preorder();
 60         }//END IF
 61     }//END preorder
 62 
 63     public void inorder() 
 64     {
 65         if (!empty())
 66         {
 67             lchild.inorder();
 68             visit();
 69             rchild.inorder();
 70         }//END IF
 71     }//END inorder
 72 
 73     public void postorder() 
 74     {
 75         if (!empty())
 76         {
 77             lchild.postorder();
 78             rchild.postorder();
 79             visit();
 80         }//END IF
 81     }//END postorder
 82 
 83     public void addLR(K key, V value, String dir) throws BinaryTreeCreationException 
 84     {
 85         try
 86         {
 87             if (dir.equals(""))
 88                 sprout(key, value);
 89             else if (dir.substring(0, 1).equalsIgnoreCase("L"))
 90                 lchild.addLR(key, value, dir.substring(1));
 91             else if (dir.substring(0, 1).equalsIgnoreCase("R"))
 92                 rchild.addLR(key, value, dir.substring(1));
 93             else
 94                 throw new BinaryTreeCreationException("### strange directional indicator: " + dir.substring(0, 1));
 95         }
 96         catch (BinaryTreeCreationException e)
 97         {
 98             throw new BinaryTreeCreationException("### no such location in the tree");
 99         }//END TRY
100     }//addLR
101 
102     public void addND(K key, V value) 
103     {
104         if (empty())
105             sprout(key, value);
106         else if (Math.random() < 0.5)
107             lchild.addND(key, value);
108         else
109             rchild.addND(key, value);
110     }//END addND
111 
112     public void addST(K key, V value) throws BinaryTreeCreationException 
113     {
114         if (empty())
115             sprout(key, value);
116         else
117         {
118             Comparable cey = (Comparable) key;
119             Comparable t = (Comparable) this.key;
120             if (cey.compareTo(t) < 0)
121                 lchild.addST(key, value);
122             else if (cey.compareTo(t) > 0)
123                 rchild.addST(key, value);
124             else
125                 throw new BinaryTreeCreationException("### can't have equality in bst");
126         }//END IF
127     }//END addST
128 
129     public V get(K key) 
130     {
131         BinaryTree<K, V> t = find(key);
132         return t.value;
133     }//END get
134 
135     public BinaryTree<K, V> find(K key) 
136     {
137         if (empty())
138             return null;
139         else if (this.key.equals(key))
140             return this;
141         else
142         {
143             BinaryTree<K, V> l = lchild.find(key);
144             if (l != null)
145                 return l;
146             BinaryTree<K, V> r = rchild.find(key);
147             if (r != null)
148                 return r;
149             return null;
150         }//END IF
151     }//END find
152 
153     public boolean member(K key) 
154     {
155         if (empty())
156             return false;
157         else if (this.key.equals(key))
158             return true;
159         else
160             return (lchild.member(key) | rchild.member(key));
161     }//END member
162 
163     public void visit() 
164     {
165         String rep;
166         if (value == null)
167             rep = "---";
168         else
169             rep = value.toString();
170         String s = "[" + key + ":" + rep + "]";
171         System.out.println(s);
172     }//END visit
173 
174     public void sprout(K key, V value) 
175     {
176         this.key = key;
177         this.value = value;
178         this.lchild = new BinaryTree<K, V>();
179         this.rchild = new BinaryTree<K, V>();
180     }//END sprout
181     
182 }//END BinaryTree

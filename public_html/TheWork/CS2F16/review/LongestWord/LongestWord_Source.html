 1 /*
 2  * The program is to prompt the user for a punctuation
 3  * free sentence with a single longest word and the print
 4  * the longest word.
 5  */
 6 package Review;
 7 
 8 import java.util.Scanner;
 9 
10 /**
11  * @author Alex
12  */
13 public class LongestWord 
14 {
15 
16     /**
17      * @param args the command line arguments
18      */
19     public static void main(String[] args) 
20     {
21         //Local Variables
22         Scanner Keyboard = new Scanner(System.in);  //Keyboard variable
23         String sent;                                //User's input sentence
24         String word;                                //Longest word
25         
26         /********** Start Main Method **********/
27         
28         //Prompt use to input 
29         System.out.print("Enter a sentence:  ");
30         sent = Keyboard.nextLine();
31         
32         //Recieve the longest word
33         word = longest(sent);
34         
35         //Display the longest word
36         System.out.println(word);
37     
38     }//END main
39 
40     private static String longest(String sent) 
41     {
42         //Local Variables
43         int maxLength = 0;                  //Highest length of words
44         String[] word = sent.split(" ");    //Words in user's sentence
45         String longWord = "";               //Longest word
46         
47         /**********Start Method**********/
48 
49         //For (every word in the sentence)
50         for (int i = 0; i < word.length; i++)
51         {
52             //IF (the word is longer than the longest word)
53             if (word[i].length() > maxLength)
54             {
55                 //Make the word the longest word
56                 maxLength = word[i].length();
57                 
58                 //Set longest word length
59                 longWord = word[i];
60             }//End IF
61             
62         }// End FOR
63         
64         //Return longest word
65         return longWord;
66     
67     }//END longest
68     
69 }//END LongestWord

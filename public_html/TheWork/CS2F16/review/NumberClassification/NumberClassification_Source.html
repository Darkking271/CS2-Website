 20         //Local Variables
 21         Random generator = new Random();    //Random number generator
 22         int num;                            //Random number
 23         int count = 0;                      //Number of random digits generated
 24         String all = "";                    //Numbers generated

 26         String even = "";                   //Even numbers

 28         String pairs = "";                  //Paired numbers
 29         
 30         /********** Start Main Method **********/
 31         
 32         //Generate random number
 33         num = generator.nextInt(900) + 100;
 34         
 35         //Count 1
 36         count++;
 37         
 38         //WHILE (count is not 25)
 39         while (count != 26)
 40         {
 41             //Add number to list of numbers
 42             all = all + num + " ";
 43             
 44             //IF (number consists of odd number)
 45             if (testOdd(num))
 46             {
 47                 //Add number to odd numbers
 48                 odd = odd + num + " ";
 49             }    
 50             //ELSE IF (number consists of even numbers)
 51             else if (testEven(num))
 52             {
 53                 //Add number to even numbers
 54                 even = even + num + " ";
 55             }
 56             //ELSE IF (number consists of mixed numbers
 57             else if (testMixed(num))
 58             {
 59                 //Add number to mixed numbers
 60                 mixed = mixed + num + " ";
 61             }
 62             //ELSE IF (number consists of pairs)
 63             else if (testPairs(num))
 64             {
 65                 //Add number to paired numbers
 66                 pairs = pairs + num + " ";
 67                 
 68             }//END IF
 69                 
 70             //Generate random number
 71             num = generator.nextInt(900) + 100;
 72         
 73             //Count 1
 74             count++;
 75             
 76         }//END WHILE
 77         
 78         //Display the generated numbers
 79         display("The Numbers...", all);

 88         display("The mixed digit numbers...", mixed);
 89

 91         display("The pairs...", pairs);

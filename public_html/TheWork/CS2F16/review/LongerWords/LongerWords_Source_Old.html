41     private static String longest(String sent) 
42     {
43         //Local Variables
44         int mean;                           //Average length of words
45         int total = 0;                      //Number of words
46         String longerWords = "";            //Words higher than average
47         String[] word = sent.split(" ");    //Words in user's sentence
48         
49         /********** Start Method **********/
50         
51         //FOR (every word in the sentence)
52         for (int i = 0; i < word.length; i++)
53         {
54             //Add the length of the word to the total
55             total += word[i].length();
56             
57         }//END FOR
58         
59         //Calculate the average length of words
60         mean = total / word.length;
61         
62         //FOR (every word in the sentence)
63         for (int i = 0; i < word.length; i++)
64         {    
65             //IF (word is longer than average)
66             if (word[i].length() > mean)
67             {    
68                 //Add word to collection of words
69                 longerWords = longerWords + word[i] + "\n";
70                 
71             }//END IF   
72             
73         }//END FOR
74         
75         //Return the longer words
76         return longerWords;
77                 
78     }//END longest
